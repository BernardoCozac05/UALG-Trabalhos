package aed.collections;

import java.util.Iterator;
import java.util.NoSuchElementException;
import java.util.function.BinaryOperator;
import java.util.function.Consumer;
import java.util.function.UnaryOperator;

/**
 * FintList  lista de inteiros com armazenamento contguo (int[]),
 * implementada como buffer circular para garantir O(1) amortizado
 * em inseres/remoes no incio e no fim, mantendo ndices lgicos
 * a comear em 0 (get(0)  o 1. elemento).
 *
 * Regras cumpridas:
 * - ndices so lgicos (o utilizador no "v" o array).
 * - addAt(0,x) e removeAt(0) -> O(1) amortizado.
 * - addAt(size,x) e remove() -> O(1) amortizado.
 * - addAt/removeAt no meio -> O(n) no pior caso (deslocaes mnimas).
 * - Iterable<Integer> com forEach(Consumer<? super Integer>).
 * - Sem colees Java; apenas int[].
 */
public class FintList implements Iterable<Integer> {

    // ---- Estado interno (buffer circular) ----
    private int[] data;   // capacidade  potncia de 2 (8,16,32,...) para usar mscara
    private int mask;     // data.length - 1 (vlido quando length  potncia de 2)
    private int head;     // ndice fsico do 1. elemento lgico
    private int size;     // nmero de elementos lgicos

    private static final int DEFAULT_CAP = 8;

    // ---- Construtor ----
    public FintList() {
        this.data = new int[DEFAULT_CAP];
        this.mask = DEFAULT_CAP - 1;
        this.head = 0;
        this.size = 0;
    }

    // ---- Utilitrios ----
    /** Mapeia ndice lgico -> ndice fsico no array. */
    private int phys(int logicalIndex) {
        return (head + logicalIndex) & mask;
    }

    /** Garante capacidade suficiente; duplica para a prxima potncia de 2. */
    private void ensureCapacity(int needed) {
        if (needed <= data.length) return;
        int cap = data.length;
        if (cap == 0) cap = DEFAULT_CAP;
        while (cap < needed) cap <<= 1; // 2x at chegar
        int[] nd = new int[cap];
        // Copiar pela ordem lgica para nd[0..size-1]
        for (int i = 0; i < size; i++) nd[i] = data[(head + i) & mask];
        data = nd;
        head = 0;
        mask = cap - 1;
    }

    private void checkNotEmpty() {
        if (size == 0) throw new IndexOutOfBoundsException("Lista vazia");
    }

    // Para addAt (0..size)
    private void checkIndexInclusive(int index) {
        if (index < 0 || index > size)
            throw new IndexOutOfBoundsException("ndice invlido: " + index);
    }

    // Para get/set/removeAt (0..size-1)
    private void checkIndexExclusive(int index) {
        if (index < 0 || index >= size)
            throw new IndexOutOfBoundsException("ndice invlido: " + index);
    }

    // ---- API pedida ----

    /** Acrescenta no fim (equivalente a addAt(size,item)). */
    public boolean add(int item) {
        ensureCapacity(size + 1);
        data[(head + size) & mask] = item; // tail
        size++;
        return true;
    }

    /** Insere em index (0..size). O(1) amortizado quando index==0 ou index==size. */
    public void addAt(int index, int item) {
        checkIndexInclusive(index);
        ensureCapacity(size + 1);

        if (index == 0) { // inserir no incio
            head = (head - 1) & mask;
            data[head] = item;
            size++;
            return;
        }
        if (index == size) { // inserir no fim
            data[(head + size) & mask] = item;
            size++;
            return;
        }

        // Insero no meio: deslocar o lado mais curto
        if (index <= size / 2) {
            // desloca prefixo [0..index-1] uma posio para a esquerda
            head = (head - 1) & mask; // cria espao lgico em 0
            for (int i = 0; i < index; i++) {
                data[(head + i) & mask] = data[(head + i + 1) & mask];
            }
            data[(head + index) & mask] = item;
        } else {
            // desloca sufixo [index..size-1] uma posio para a direita
            for (int i = size; i > index; i--) {
                data[(head + i) & mask] = data[(head + i - 1) & mask];
            }
            data[(head + index) & mask] = item;
        }
        size++;
    }

    /** Retorna o elemento em index (0..size-1). */
    public int get(int index) {
        checkIndexExclusive(index);
        return data[phys(index)];
    }

    /** Retorna o 1. elemento. */
    public int getFirst() {
        checkNotEmpty();
        return data[head];
    }

    /** Retorna o ltimo elemento. */
    public int get() {
        checkNotEmpty();
        return data[(head + size - 1) & mask];
    }

    /** Altera o elemento em index. */
    public void set(int index, int value) {
        checkIndexExclusive(index);
        data[phys(index)] = value;
    }

    /** true se vazia. */
    public boolean isEmpty() { return size == 0; }

    /** n de elementos. */
    public int size() { return size; }

    /** Remove e devolve o ltimo elemento. */
    public int remove() {
        checkNotEmpty();
        int tailIdx = (head + size - 1) & mask;
        int v = data[tailIdx];
        size--;
        return v;
    }

    /**
     * Remove e devolve o elemento na posio index.
     * O(1) quando index==0 (incio) ou index==size-1 (fim).
     * O(n) no meio (desloca o lado mais curto).
     */
    public int removeAt(int index) {
        checkIndexExclusive(index);

        if (index == 0) { // remover do incio
            int v = data[head];
            head = (head + 1) & mask;
            size--;
            return v;
        }
        if (index == size - 1) { // remover do fim
            int v = data[(head + size - 1) & mask];
            size--;
            return v;
        }

        int removed = data[phys(index)];
        if (index <= size / 2) {
            // desloca prefixo [0..index-1] uma posio para a direita
            for (int i = index; i > 0; i--) {
                data[phys(i)] = data[phys(i - 1)];
            }
            head = (head + 1) & mask; // consumiu uma clula do incio
        } else {
            // desloca sufixo [index+1..size-1] uma posio para a esquerda
            for (int i = index; i < size - 1; i++) {
                data[phys(i)] = data[phys(i + 1)];
            }
        }
        size--;
        return removed;
    }

    /** Remove a primeira ocorrncia de item. */
    public boolean remove(int item) {
        int idx = indexOf(item);
        if (idx == -1) return false;
        removeAt(idx);
        return true;
    }

    /** true se contiver item. */
    public boolean contains(int item) { return indexOf(item) != -1; }

    /** ndice da primeira ocorrncia ou -1. */
    public int indexOf(int item) {
        for (int i = 0; i < size; i++) if (data[phys(i)] == item) return i;
        return -1;
    }

    /** Inverte a lista (in-place). */
    public void reverse() {
        for (int i = 0, j = size - 1; i < j; i++, j--) {
            int pi = phys(i), pj = phys(j);
            int tmp = data[pi]; data[pi] = data[pj]; data[pj] = tmp;
        }
    }

    /** Cpia profunda (array independente). */
    public FintList deepCopy() {
        FintList c = new FintList();
        c.ensureCapacity(this.size);
        for (int i = 0; i < this.size; i++) c.data[i] = this.data[(this.head + i) & this.mask];
        c.size = this.size;
        c.head = 0;
        return c;
    }

    // ---- Iterao ----
    @Override
    public Iterator<Integer> iterator() { return new It(); }

    private final class It implements Iterator<Integer> {
        private int i = 0;
        @Override public boolean hasNext() { return i < size; }
        @Override public Integer next() {
            if (!hasNext()) throw new NoSuchElementException();
            return data[phys(i++)];
        }
    }

    /** Assinatura correta por Iterable<T>. */
    @Override
    public void forEach(Consumer<? super Integer> c) {
        for (int i = 0; i < size; i++) c.accept(data[phys(i)]);
    }

    /** Aplica op a cada elemento, substituindo-o. */
    public void map(UnaryOperator<Integer> op) {
        for (int i = 0; i < size; i++) data[phys(i)] = op.apply(data[phys(i)]);
    }

    /** Acumula todos os elementos com op, a partir de defaultValue. */
    public int reduce(BinaryOperator<Integer> op, int defaultValue) {
        int acc = defaultValue;
        for (int i = 0; i < size; i++) acc = op.apply(acc, data[phys(i)]);
        return acc;
    }
