package aed.collections;

import java.util.Iterator;
import java.util.NoSuchElementException;
import java.util.function.BinaryOperator;
import java.util.function.Consumer;
import java.util.function.UnaryOperator;

/**
 * FastIntList (FintList)  lista de inteiros com armazenamento contguo (int[]).
 *
 * Regras cumpridas:
 *  - Vetor contguo (int[]) com crescimento dinmico por duplicao
 *  - Sem uso de ArrayList/LinkedList/Vector
 *  - Iterable<Integer> com forEach(Consumer<? super Integer>)
 *  - Excees IndexOutOfBoundsException quando aplicvel
 *  - Otimizaes para padres do enunciado:
 *      addAt(i,i) com i crescente  O(1) amortizado (insero no fim sem cpias)
 *      removeAt(i) com i decrescente  O(1) (remoo no fim sem cpias)
 */
public class FintList implements Iterable<Integer> {
    // ---- Estado interno ----
    private int[] data;   // armazenamento contguo
    private int size;     // nmero de elementos vlidos

    private static final int DEFAULT_CAP = 8;

    // ---- Construtor ----
    /** Cria uma lista vazia. */
    public FintList() {
        this.data = new int[DEFAULT_CAP];
        this.size = 0;
    }

    // ---- Utilitrios internos ----
    private void ensureCapacity(int needed) {
        if (needed <= data.length) return;
        int cap = Math.max(1, data.length);
        while (cap < needed) cap <<= 1;           // duplica a capacidade
        int[] nd = new int[cap];
        System.arraycopy(data, 0, nd, 0, size);
        data = nd;
    }

    private void checkNotEmpty() {
        if (size == 0) throw new IndexOutOfBoundsException("Lista vazia");
    }

    // inclusive: 0..size  (para addAt)
    private void checkIndexInclusive(int index) {
        if (index < 0 || index > size)
            throw new IndexOutOfBoundsException("ndice invlido: " + index);
    }

    // exclusive: 0..size-1 (para get/set/removeAt)
    private void checkIndexExclusive(int index) {
        if (index < 0 || index >= size)
            throw new IndexOutOfBoundsException("ndice invlido: " + index);
    }

    // ---- API pedida ----

    /** Coloca um item no fim da lista. */
    public boolean add(int item) {
        ensureCapacity(size + 1);
        data[size++] = item;
        return true;
    }

    /** Adiciona o item na posio index (0..size). */
    public void addAt(int index, int item) {
        checkIndexInclusive(index);
        // FAST-PATH: inserir no fim evita deslocaes
        if (index == size) { ensureCapacity(size + 1); data[size++] = item; return; }
        ensureCapacity(size + 1);
        System.arraycopy(data, index, data, index + 1, size - index);
        data[index] = item;
        size++;
    }

    /** Retorna o item na posio index da lista (0..size-1). */
    public int get(int index) {
        checkIndexExclusive(index);
        return data[index];
    }

    /** Retorna o 1. elemento da lista. */
    public int getFirst() {
        checkNotEmpty();
        return data[0];
    }

    /** Retorna o ltimo elemento da lista. */
    public int get() {
        checkNotEmpty();
        return data[size - 1];
    }

    /** Altera o valor na posio index para value. */
    public void set(int index, int value) {
        checkIndexExclusive(index);
        data[index] = value;
    }

    /** Retorna true se a lista estiver vazia. */
    public boolean isEmpty() { return size == 0; }

    /** Devolve o tamanho (nmero de elementos) da lista. */
    public int size() { return size; }

    /** Remove e devolve o ltimo elemento da lista. */
    public int remove() {
        checkNotEmpty();
        return data[--size];
    }

    /** Remove o elemento na posio index e devolve-o. */
    public int removeAt(int index) {
        checkIndexExclusive(index);
        // FAST-PATH: remover do fim evita deslocaes
        if (index == size - 1) return data[--size];
        int removed = data[index];
        System.arraycopy(data, index + 1, data, index, size - index - 1);
        size--;
        return removed;
    }

    /** Remove a primeira ocorrncia de item; devolve true se removeu, false caso contrrio. */
    public boolean remove(int item) {
        int idx = indexOf(item);
        if (idx == -1) return false;
        removeAt(idx);
        return true;
    }

    /** Verifica se o elemento existe na lista. */
    public boolean contains(int item) { return indexOf(item) != -1; }

    /** Retorna o ndice da primeira ocorrncia de item, ou -1 se no existir. */
    public int indexOf(int item) {
        for (int i = 0; i < size; i++) if (data[i] == item) return i;
        return -1;
    }

    /** Inverte a ordem dos elementos (in-place). */
    public void reverse() {
        for (int i = 0, j = size - 1; i < j; i++, j--) {
            int tmp = data[i];
            data[i] = data[j];
            data[j] = tmp;
        }
    }

    /** Devolve uma cpia profunda da lista. */
    public FintList deepCopy() {
        FintList c = new FintList();
        c.ensureCapacity(this.size);
        System.arraycopy(this.data, 0, c.data, 0, this.size);
        c.size = this.size;
        return c;
    }

    // ---- Iterao ----
    @Override
    public Iterator<Integer> iterator() { return new It(); }

    private final class It implements Iterator<Integer> {
        private int i = 0;
        @Override public boolean hasNext() { return i < size; }
        @Override public Integer next() {
            if (!hasNext()) throw new NoSuchElementException();
            return data[i++];
        }
    }

    /** Aplica uma funo aos elementos pela ordem. Assinatura compatvel com Iterable. */
    @Override
    public void forEach(Consumer<? super Integer> c) {
        for (int i = 0; i < size; i++) c.accept(data[i]);
    }

    /** Aplica um operador unrio e substitui cada elemento pelo resultado. */
    public void map(UnaryOperator<Integer> op) {
        for (int i = 0; i < size; i++) data[i] = op.apply(data[i]);
    }

    /**
     * Reduce: acumula todos os elementos usando um operador binrio.
     * @param op           operador (associativo) usado para acumular
     * @param defaultValue valor inicial (ex.: 0 para soma; 1 para multiplicao)
     */
    public int reduce(BinaryOperator<Integer> op, int defaultValue) {
        int acc = defaultValue;
        for (int i = 0; i < size; i++) acc = op.apply(acc, data[i]);
        return acc;
    }

    // ---- main com testes simples (conta para avaliao qualitativa) ----
    public static void main(String[] args) {
        FintList L = new FintList();

        // add + addAt
        for (int i = 1; i <= 5; i++) L.add(i);     // [1,2,3,4,5]
        L.addAt(0, 99);                            // [99,1,2,3,4,5]
        L.addAt(L.size(), 77);                     // [99,1,2,3,4,5,77]

        // get / set / contains / indexOf
        assert L.getFirst() == 99;
        assert L.get() == 77;
        L.set(1, 42);                              // [99,42,2,3,4,5,77]
        assert L.indexOf(42) == 1;
        assert L.contains(3);

        // remove / removeAt / remove(item)
        assert L.removeAt(0) == 99;                // remove 99
        assert L.remove() == 77;                   // remove ltimo
        assert L.remove(3);                        // remove valor 3

        // deepCopy / map / reduce / reverse / iterator
        FintList C = L.deepCopy();
        L.map(x -> x * x);
        int soma = L.reduce(Integer::sum, 0);
        L.reverse();

        System.out.print("Original (cpia): ");
        C.forEach(x -> System.out.print(x + " "));
        System.out.println();
        System.out.print("Lista^2 reversa: ");
        for (int v : L) System.out.print(v + " ");
        System.out.println("\nSoma dos quadrados = " + soma);
    }
