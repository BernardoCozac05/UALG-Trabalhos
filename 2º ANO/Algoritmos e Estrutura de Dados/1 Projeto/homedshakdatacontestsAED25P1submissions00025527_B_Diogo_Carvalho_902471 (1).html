import java.io.BufferedReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Iterator;
import java.util.List;

public class Scene {

    public enum Type { NORMAL, WON, FAILED }

    private final int id;
    private final String description;
    private final Type type;
    private final ArrayList<Option> options;

    public Scene(int id, String description, Type type) {
        if (description == null) {
            throw new IllegalArgumentException("description cannot be null");
        }
        if (type == null) {
            throw new IllegalArgumentException("type cannot be null");
        }
        this.id = id;
        this.description = description;
        this.type = type;
        this.options = new ArrayList<>();
    }

    /**
     * Adiciona uma opção no fim da lista se não existir outra com o mesmo id.
     * Para cenas WON/FAILED não adiciona (retorna false).
     */
    public boolean addOption(Option o) {
        if (o == null || this.type != Type.NORMAL) return false;
        int newId = o.getId();
        for (Option cur : options) {
            if (cur.getId() == newId) return false; // conflito
        }
        options.add(o);
        return true;
    }

    public int getId() { return id; }

    public Type getType() { return type; }

    public String getDescription() { return description; }

    /** Coleção iterável (não-modificável) das opções. */
    public Iterable<Option> getOptions() {
        final List<Option> ro = Collections.unmodifiableList(options);
        // devolver Iterable simples, como pede o enunciado
        return new Iterable<Option>() {
            @Override public Iterator<Option> iterator() { return ro.iterator(); }
        };
    }

    /**
     * Formato textual:
     * [id]
     * <descrição>
     * + x. desc
     * ...
     * (ou <WON>/<FAILED> para cenas finais)
     */
    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder();
        sb.append('[').append(id).append(']').append('\n');
        sb.append(description).append('\n');
        if (type == Type.WON) {
            sb.append("<WON>").append('\n');
        } else if (type == Type.FAILED) {
            sb.append("<FAILED>").append('\n');
        } else {
            for (Option op : options) {
                sb.append(op.toString()).append('\n');
            }
        }
        return sb.toString();
    }

    /**
     * Lê uma cena do BufferedReader aberto, no formato do enunciado:
     * NORMAL:
     *   [id]
     *   <<<
     *   (várias linhas de descrição)
     *   >>>
     *   ***
     *   (opções, 0..n linhas)
     *   ***
     * WON:
     *   [id]
     *   <<<
     *   (várias linhas)
     *   >>>WON
     * FAILED:
     *   [id]
     *   <<<
     *   (várias linhas)
     *   >>>FAILED
     *
     * Não fecha o reader. Lança IOException / NumberFormatException.
     */
    public static Scene parseScene(BufferedReader br) throws IOException, NumberFormatException {
        // 1) [id]
        String line = br.readLine();
        if (line == null) throw new IOException("unexpected EOF reading scene id");
        line = line.trim();
        if (line.length() < 2 || line.charAt(0) != '[' || line.charAt(line.length()-1) != ']') {
            throw new NumberFormatException("invalid scene id line");
        }
        String idStr = line.substring(1, line.length()-1).trim();
        int id = Integer.parseInt(idStr);

        // 2) <<<
        line = br.readLine();
        if (line == null) throw new IOException("unexpected EOF reading <<<");
        if (!line.trim().equals("<<<")) {
            throw new IOException("expected <<<");
        }

        // 3) descrição até >>>, >>>WON ou >>>FAILED
        StringBuilder desc = new StringBuilder();
        Type type = Type.NORMAL;

        while (true) {
            line = br.readLine();
            if (line == null) throw new IOException("unexpected EOF reading description");

            String trimmed = line.trim();

            if (trimmed.equals(">>>")) {
                // cena NORMAL
                type = Type.NORMAL;
                break;
            } else if (trimmed.equals(">>>WON")) {
                type = Type.WON;
                break;
            } else if (trimmed.equals(">>>FAILED")) {
                type = Type.FAILED;
                break;
            } else {
                // acumula a linha tal como está
                if (desc.length() > 0) desc.append('\n');
                desc.append(line);
            }
        }

        Scene scene = new Scene(id, desc.toString(), type);

        // Para cenas finais, não existem opções. Termina já.
        if (type != Type.NORMAL) {
            return scene;
        }

        // 4) *** (início das opções)
        line = br.readLine();
        if (line == null) throw new IOException("unexpected EOF reading options start");
        if (!line.trim().equals("***")) {
            throw new IOException("expected *** (options start)");
        }

        // 5) 0..n opções, terminadas por *** (fim)
        while (true) {
            line = br.readLine();
            if (line == null) throw new IOException("unexpected EOF reading options");
            String trimmed = line.trim();
            if (trimmed.equals("***")) {
                break; // fim das opções
            }
            // cada linha é uma opção no formato do Option.toString()
            Option op = Option.parseOption(line);
            scene.addOption(op); // respeita conflito de ids
        }

        return scene;
    }
}
