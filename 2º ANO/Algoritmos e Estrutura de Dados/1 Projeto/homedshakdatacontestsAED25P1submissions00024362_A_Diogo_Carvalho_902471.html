public class Option {

    private final int id;
    private final String description;

    /**
     * Cria uma opção com a informação recebida.
     * @param id identificador da opção (não pode ser negativo)
     * @param description descrição textual (não pode ser nula nem vazia)
     * @throws IllegalArgumentException se id < 0 ou descrição inválida
     */
    public Option(int id, String description) {
        if (id < 0) {
            throw new IllegalArgumentException("id cannot be negative");
        }
        if (description == null || description.trim().isEmpty()) {
            throw new IllegalArgumentException("description cannot be null or empty");
        }
        this.id = id;
        this.description = description;
    }

    public int getId() { return id; }

    public String getDescription() { return description; }

    /** Duas opções são iguais se tiverem o mesmo id e a mesma descrição. */
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Option)) return false;
        Option other = (Option) o;
        return this.id == other.id && this.description.equals(other.description);
    }

    @Override
    public int hashCode() {
        int result = Integer.hashCode(id);
        result = 31 * result + description.hashCode();
        return result;
    }

    /** Formato textual: "+ <id>. <description>" */
    @Override
    public String toString() {
        return "+ " + id + ". " + description;
    }

    /**
     * Lê uma Option a partir de uma string no formato do toString().
     * Aceita espaços variáveis. Se o id não for inteiro, a NumberFormatException
     * terá a mensagem padrão do Java (ex.: For input string: "x").
     *
     * @param s linha no formato "+ <id>. <desc>"
     * @return Option
     * @throws NumberFormatException se o formato não casar ou id não for inteiro
     * @throws IllegalArgumentException se a descrição extraída for inválida
     */
    public static Option parseOption(String s) {
        if (s == null) {
            throw new NumberFormatException("null string");
        }
        String str = s.trim();

        // Captura:
        //  grupo 1 -> "id" como QUALQUER token não-espaço; o parseInt decide se é número
        //  grupo 2 -> descrição (pelo menos um char)
        java.util.regex.Pattern p = java.util.regex.Pattern.compile("^\\+\\s*(\\S+)\\s*\\.\\s*(.+)\\s*$");
        java.util.regex.Matcher m = p.matcher(str);

        if (!m.matches()) {
            // Não está no formato "+ <algo>. <desc>"
            throw new NumberFormatException("Invalid option format or id");
        }

        int parsedId = Integer.parseInt(m.group(1)); // pode lançar NFE com msg padrão
        String desc = m.group(2);

        return new Option(parsedId, desc);
    }
}
