// Scene.java — Projeto 1 – Problema B (finalíssimo)
// (sem package)

import java.io.BufferedReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Iterator;
import java.util.List;

public class Scene {

    public enum Type { NORMAL, WON, FAILED }

    private final int id;
    private final String description;
    private final Type type;
    private final ArrayList<Option> options;

    public Scene(int id, String description, Type type) {
        if (description == null) throw new IllegalArgumentException("description cannot be null");
        if (type == null)        throw new IllegalArgumentException("type cannot be null");
        this.id = id;
        this.description = description;
        this.type = type;
        this.options = new ArrayList<>();
    }

    public int getId() { return id; }
    public String getDescription() { return description; }
    public Type getType() { return type; }

    /** Iterable somente-leitura das opções. */
    public Iterable<Option> getOptions() {
        final List<Option> ro = Collections.unmodifiableList(options);
        return new Iterable<Option>() {
            @Override public Iterator<Option> iterator() { return ro.iterator(); }
        };
    }

    /** Só adiciona em NORMAL e recusa ids repetidos. */
    public boolean addOption(Option o) {
        if (o == null || type != Type.NORMAL) return false;
        int nid = o.getId();
        for (Option p : options) if (p.getId() == nid) return false;
        options.add(o);
        return true;
    }

    /**
     * Formato EXACTO esperado:
     * [id]\n
     * <descrição>\n
     * (se NORMAL) linhas "id. descrição"\n
     * Termina SEMPRE com '\n' (os testes contam linhas).
     * Em cenas finais imprime só a descrição (sem marcadores).
     */
    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder(64);
        sb.append('[').append(id).append(']').append('\n');
        sb.append(description).append('\n'); // descrição termina com \n

        if (type != Type.NORMAL) {
            return sb.toString(); // finais: só descrição + \n
        }

        for (int i = 0; i < options.size(); i++) {
            Option op = options.get(i);
            sb.append(op.getId()).append(". ").append(op.getDescription());
            sb.append('\n'); // cada opção numa linha; \n final é esperado nos testes
        }
        return sb.toString();
    }

    /**
     * Parser:
     * [id]
     * <<<
     * (linhas descrição)
     * >>> | >>>WON | >>>FAILED
     * se NORMAL:
     * ***
     * + <id>. <desc>   OU   <id>. <desc>
     * ...
     * ***
     * Não fecha o reader.
     */
    public static Scene parseScene(BufferedReader br) throws IOException, NumberFormatException {
        String line;

        // --- [id] (ignora linhas vazias antes) ---
        do {
            line = br.readLine();
            if (line == null) throw new IOException("unexpected EOF reading scene id");
            line = line.trim();
        } while (line.isEmpty());

        if (line.length() < 2 || line.charAt(0) != '[' || line.charAt(line.length()-1) != ']')
            throw new NumberFormatException("invalid scene id line");
        int id = Integer.parseInt(line.substring(1, line.length()-1).trim());

        // --- <<< ---
        line = br.readLine();
        if (line == null) throw new IOException("unexpected EOF reading <<<");
        if (!line.trim().equals("<<<")) throw new IOException("expected <<<");

        // --- descrição até >>> / >>>WON / >>>FAILED ---
        StringBuilder desc = new StringBuilder();
        Type type = Type.NORMAL;
        while (true) {
            line = br.readLine();
            if (line == null) throw new IOException("unexpected EOF reading description");
            String t = line.trim();
            if (t.equals(">>>"))       { type = Type.NORMAL;  break; }
            if (t.equals(">>>WON"))    { type = Type.WON;     break; }
            if (t.equals(">>>FAILED")) { type = Type.FAILED;  break; }
            if (desc.length() > 0) desc.append('\n');
            desc.append(line);
        }

        Scene scene = new Scene(id, desc.toString(), type);
        if (type != Type.NORMAL) return scene; // finais não têm opções

        // --- *** (abre opções) ---
        // (ignora linhas vazias intermédias por robustez)
        do {
            line = br.readLine();
            if (line == null) throw new IOException("unexpected EOF reading options start");
            line = line.trim();
        } while (line.isEmpty());
        if (!line.equals("***")) throw new IOException("expected *** (options start)");

        // --- opções até *** ---
        while (true) {
            line = br.readLine();
            if (line == null) throw new IOException("unexpected EOF reading options");
            String raw = line;
            String t = raw.trim();
            if (t.equals("***")) break; // fim

            // 1) tenta "+ <id>. <desc>"
            try {
                if (t.startsWith("+")) {
                    scene.addOption(Option.parseOption(raw));
                    continue;
                }
            } catch (NumberFormatException ignore) {
                // cai para o formato simples
            }

            // 2) aceita também "<id>. <desc>"
            int dot = raw.indexOf('.');
            if (dot <= 0) throw new NumberFormatException("invalid option line");
            String idStr = raw.substring(0, dot).trim();
            int optId = Integer.parseInt(idStr); // pode lançar NFE (mensagem padrão)
            String descStr = raw.substring(dot + 1).trim();

            // remover um ponto final terminal, se existir
            if (!descStr.isEmpty() && descStr.charAt(descStr.length()-1) == '.') {
                descStr = descStr.substring(0, descStr.length()-1).trim();
            }

            scene.addOption(new Option(optId, descStr));
        }

        return scene;
    }
}
